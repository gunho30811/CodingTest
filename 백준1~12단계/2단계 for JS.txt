1. 두 수 비교하기
일반버젼
const fs=require('fs');
const input=fs.readFileSync("/dev/stdin").toString().split(" ");
const A=+(input[0]);
const B=+(input[1]);
if(A>B){console.log(">");}
else if(A<B){console.log("<");}
else {console.log("==");}

삼항연산자 버젼
let[A,B]=require('fs').readFileSync("/dev/stdin").toString().split(" ");
A=+A;
B=+B;
(A>B?console.log(">"):(A<B ? console.log("<") : console.log("==")));
